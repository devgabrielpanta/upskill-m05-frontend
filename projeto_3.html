📡 Exemplo — apiTaskService

Ficheiro: api/apiTaskService.ts

import { Task } from '../models/task';

const BASE_URL = 'http://localhost:3000';

// GET /tasks
export async function getTasks(): Promise<Task[]> {
  const res = await fetch(`${BASE_URL}/tasks`);

  if (!res.ok) {
    throw new Error('Erro ao buscar tasks');
  }

  return await res.json();
}

// PUT /tasks/:id
export async function updateTask(
  id: number,
  task: Task
): Promise<Task> {

  const res = await fetch(`${BASE_URL}/tasks/${id}`, {
    method: 'PUT',
    headers: {
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(task)
  });

  if (!res.ok) {
    throw new Error('Erro ao atualizar task');
  }

  return await res.json();
}

🔄 Integração com Array no Frontend

Mesmo usando API, continuamos a usar um array local para renderizar os dados.

import { getTasks } from '../api/apiTaskService';
import { Task } from '../models/task';

// estado do frontend
let tasks: Task[] = [];

📌 Função loadTasks (ESSENCIAL)

async function loadTasks() {
  tasks = await getTasks();

  renderTasks();
}

📌 Renderização

function renderTasks() {
  console.log(tasks); // ou atualizar HTML
}

📌 Operações com sincronização

// criar task
async function createTask(task: Task) {
  await apiCreateTask(task);

  await loadTasks();
}

// atualizar task
async function editTask(task: Task) {
  await updateTask(task.id, task);

  await loadTasks();
}

// apagar task
async function deleteTask(id: number) {
  await apiDeleteTask(id);

  await loadTasks();
}

🧠 Regra mais importante

API → dados reais
Array → estado do frontend
loadTasks() → sincroniza os dois

Após qualquer POST, PUT ou DELETE:

await loadTasks();
